        allMdx(
            sort: { fields: [frontmatter___date], order: DESC }
            filter: { frontmatter: { published: { eq: true } } }
            limit: 3
        ) {
            edges {
                node {
                    id
                    frontmatter {
                        title
                        date
                        path
                        preview
                        description
                        images {

                                        ...GatsbyImageSharpFluid_tracedSVG

                                }
                            }
                            is_cover
